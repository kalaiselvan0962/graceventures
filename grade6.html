import pygame
import sys

# 1. Initialize Pygame
pygame.init()

# 2. Set up game grid values
ROWS, COLS = 10, 10
BLOCK_SIZE = 40  # Each block is 40px wide

# 3. Calculate game window size
WIDTH, HEIGHT = COLS * BLOCK_SIZE, ROWS * BLOCK_SIZE
TOP_MARGIN = 40  # Extra space at the top for score or text

# 4. Create the actual game window
screen = pygame.display.set_mode((WIDTH, HEIGHT + TOP_MARGIN))

# 5. Set the window title
pygame.display.set_caption("Falling Blocks")

# 6. Create a clock to control FPS
clock = pygame.time.Clock()

# 7. Function to draw the grid lines
def draw_grid():
    for row in range(ROWS):
        for col in range(COLS):
            pygame.draw.rect(
                screen,
                (40, 40, 40),  # Dark gray grid lines
                (col * BLOCK_SIZE, row * BLOCK_SIZE + TOP_MARGIN, BLOCK_SIZE, BLOCK_SIZE),
                1  # Border thickness
            )

# 8. Game loop
running = True
while running:
    for event in pygame.event.get():
        # If user clicks close button, exit game
        if event.type == pygame.QUIT:
            running = False

    # Fill background black
    screen.fill((0, 0, 0))

    # Draw the grid
    draw_grid()

    # Update the display
    pygame.display.update()

    # Control game speed (60 FPS)
    clock.tick(60)

# Quit Pygame safely
pygame.quit()
sys.exit()
